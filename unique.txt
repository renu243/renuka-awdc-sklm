#include <stdio.h>
int main() {
   int a[5];
int i,j;
int temp=0;
for(i=0;i<5;i++)
{
scanf("%d",&a[i]);
}

for(i=0;i<5;i++)
{
for(j=i+1;j<5;j++)
{
if(a[i]==a[j])
temp++;
}
}
if(temp==0)
printf("unique");
else
printf("duplicate");
return 0;
}
